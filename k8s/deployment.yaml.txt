apiVersion: apps/v1
kind: Deployment
metadata:
  name: devops-demo-app
  labels:
    app: devops-demo
spec:
  replicas: 2  # Run 2 copies of your app for redundancy
  selector:
    matchLabels:
      app: devops-demo
  template:
    metadata:
      labels:
        app: devops-demo
    spec:
      containers:
      - name: app
        image: your-dockerhub-username/devops-demo:latest  # CHANGE THIS
        ports:
        - containerPort: 8080
        env:
        - name: PORT
          value: "8080"
        # Health check that Kubernetes uses
        livenessProbe:
          httpGet:
            path: /health
            port: 8080
          initialDelaySeconds: 5
          periodSeconds: 10
        readinessProbe:
          httpGet:
            path: /health
            port: 8080
          initialDelaySeconds: 3
          periodSeconds: 5
